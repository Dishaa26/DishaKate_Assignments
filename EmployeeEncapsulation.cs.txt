using System;

class Employee
{
    private int id;
    private string name;
    private double salary;

    public int Id
    {
        get { return id; }
        set { id = value; }
    }

    public string Name
    {
        get { return name; }
        set { name = value; }
    }

    public double Salary
    {
        get { return salary; }
        set { salary = value; }
    }
}

class Program
{
    static void Main()
    {
        Employee emp = new Employee();
        emp.Id = 1;
        emp.Name = "Shila";
        emp.Salary = 55000;

        Console.WriteLine($"{emp.Id} - {emp.Name} - {emp.Salary}");
    }
}
